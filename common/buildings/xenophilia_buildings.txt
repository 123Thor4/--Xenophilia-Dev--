#Authoritarian
breeding_clinic = {
	base_buildtime = 360
	base_cap_amount = 1

	category = amenity
	
	potential = {
		owner = { 
			is_regular_empire = yes 
			has_valid_civic = civic_pop_quotas
		}
		NOT = { has_modifier = slave_colony }
	}	

	destroy_trigger = {
		exists = owner
		OR = {
			owner = { is_regular_empire = no }
			has_modifier = slave_colony
			AND = {
				buildings_stop_pop = yes
				owner = { is_ai = yes }
			}
		}
	}
	
	planet_modifier = {
		job_breeder_add = 2
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 300
		}
		upkeep = {
			energy = 2
		}
	}
	
	prerequisites = {
		
	}
	
	upgrades = {
		"breeding_centre"
	}
	
	triggered_desc = {
		text = job_breeding_clinic_effect_desc
	}
	
	ai_weight = {
		weight = 75000
		modifier = {
			factor = 0
			OR = {
				is_planet_class = pc_habitat
				buildings_stop_pop = yes
			}
		}
		modifier = {
			factor = 0
			count_pops = { limit = { is_sapient = yes } count > 75 }
		}
	}
}
